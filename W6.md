#include <iostream>

int main() { 
  int n;
  scanf("%d",&n);
  int a[11];
  a[0] = 1;
  a[1] = 1;
  a[2] = 2;
  for(int i = 3; i <= 11; i++)
    {
      a[i] = a[i-1] + a[i-2] + a[i-3];
    }
  
  for(int i = 0; i < n; i++)
    {
      int f;
      scanf("%d",&f);
      printf("%d\n", a[f]);
    }
}
