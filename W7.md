#include <iostream>
#include <iostream>
#include <algorithm>
#include <cstdio>
#include <iostream>
#include <utility>
#include <stack>

int main() {
  int a,b,d,n;
  scanf("%d",&a);
  scanf("%d",&b);
  scanf("%d",&d);
  scanf("%d",&n);

  int h[10000];
  h[0] = 1;
  for(int i = 1;i<=n;i++)
    {
      h[i] = h[i-1];
      if(i>= a)
        h[i] += h[i-a];
      if(i>= b)
        h[i] -= h[i-b];
    }
  if(n-d >= 0)
    printf("%d",(h[n] - h[n-d] + 1000) % 1000);
  else
    printf("%d", h[n]);
  
  
}
