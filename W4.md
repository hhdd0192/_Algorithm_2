#include <cstdio>
#include <algorithm>
#include <vector>
#include <iostream>
#include <utility>
#include <stack>
using namespace std;


int main()
{
  int n,m; int x[10000]; int y[10000];
  int x1, y1; 
  
  scanf("%d %d",&n, &m);
  
  
  for(int i = 0; i < m; i++)
    scanf("%d %d", &x[i], &y[i]);
  
  sort(x,x+m);
  sort(y,y+m);
  x1 = x[m/2];
  y1 = y[m/2];

  
  int sum = 0;
  for(int  i = 0; i < m; i++)
    {
      sum += abs(x1 - x[i]) + abs(y1 - y[i]);
    }
  printf("%d", sum);

    
  //abs(~~)
  


  
}
