#include <cstdio>
#include <algorithm>
#include <vector>
#include <iostream>
#include <utility>
#include <stack>
using namespace std;
#define inside(a,b,c) ((a) <= (c) && (c) <= (b))

int main()
{
  int x[10001],y[10001];
  int size, l,f;
  scanf("%d %d %d", &size,&l,&f);

  for(int i = 0; i < f;i++)
    {
      scanf("%d %d", &x[i],&y[i]);
    }

  int a = l/2;
  
  int h;
  int x1,y1;
  int max = 0;
  for(int w = 1; w < a; w ++)
    {
      h = a-w;
      for(int j = 0; j < f; j++)
        {
          //j번 위치 점 고르기 x[j], y[j]
          x1 = x[j];
          y1 = y[j];
          int cnt = 0;
          for(int k = 0; k < f; k++)
            {
              if(inside(x1,x1 + w,x[k]) && inside(y1,y1 + h,y[k]))
              {
                cnt++;
              }
            }
          if(max < cnt)
            max = cnt;
        }
    }

  printf("%d",max);

  
  

  
}
