
M,N 사이의 소수 구하기
```c
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int main() {
  int m,n;
  int f;
  scanf("%d %d", &m,&n);

  for(int i = m; i < n; i++)
    {
      f = 0;
      for(int j = 2; j*j<i ; j++)
        {
          if(i % j == 0)
            f = 1;
        }
      if(f == 0)
        printf("%d\n", i);
    }
}
```
