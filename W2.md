#include <cstdio>
#include <algorithm>
#include <vector>
#include <iostream>
#include <utility>

std::vector <int> v;
std::vector <int> temp;

void print_vector_all(int n)
{
	for(int i = 0; i < n; i++)
	{
		std::cout.width(2);
		std::cout << v[i] << ' ';
	}
	std::cout << '\n';

	for(int i = 0; i< n; i++)
	{
		std::cout.width(2);
		std:: cout << temp[i] << ' ';
	}
	std::cout << '\n';
}

int main(void)
{
	int n, k;
	scanf("%d %d", &n, &k);
	
	for(int i = 1; i <= n; i++)
	{
		v.push_back(i);
	}
	for(int i = 0; i< n-k; i++)
	{
		temp.push_back(0);
	}
	for(int i = 0; i < k; i++)
	{
		temp.push_back(1);
	}

	std::cout << "원본\n";
	print_vector_all(n);	
	
	do{
		print_vector_all(n);
		for(int i = 0; i < n; i++)
		{
			if(temp[i] == 1)
			{
				std::cout.width(2);
				std::cout << v[i] << ' ';
			}
		}
		std::cout << '\n'<< '\n';

	}while(std::next_permutation(temp.begin(), temp.end())); 
	return 0;
}
